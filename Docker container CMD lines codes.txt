# To pull and run the image locally or from Docker Hub
docker run -p 5000:5000 --name cnn_api_container -d khaikit/govtech_va_assessment 

# API url to ping: "http://127.0.0.1:5000/"
# API url to infer: "http://127.0.0.1:5000/predict"

# To stop the container
docker stop cnn_app_container

# To remove the container
docker rm cnn_app_container

# To remove the image
docker image rm khaikit/govtech_va_assessment
